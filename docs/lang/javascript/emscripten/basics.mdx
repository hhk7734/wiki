## 설치

```shell
git clone https://github.com/emscripten-core/emsdk.git \
&& cd emsdk
```

```shell
./emsdk install 3.0.0
```

```shell
./emsdk activate 3.0.0
```

```shell
source ./emsdk_env.sh
```

## Setup

```shell
<package>/
├── CMakeLists.txt
├── c_src/
│   ├── funcs.cpp
│   ├── funcs.h
│   └── <target>.cpp
└── test/
    └── ...
```

```cmake title="CMakeLists.txt"
cmake_minimum_required(VERSION 3.15)
set(PROJECT_NAME "<target>")
project(${PROJECT_NAME} LANGUAGES CXX)

set(TARGET_NAME "${PROJECT_NAME}")

set(SOURCE
    "c_src/funcs.cpp"
    "c_src/<target>.cpp"
)

add_executable(${TARGET_NAME} ${SOURCE})

target_link_options(${TARGET_NAME} PUBLIC "--bind")
```

## Wrapper

```cpp title="c_src/funcs.h"
#pragma once

#include <string>

std::string hello(void);
int         add(int a, int b);
```

```cpp title="c_src/funcs.cpp"
#include "funcs.h"

std::string hello(void) { return "Hello World!"; }

int add(int a, int b) { return a + b; }
```

```cpp title="c_src/<target>.cpp"
#include "funcs.h"

#include <emscripten/bind.h>

using namespace emscripten;

EMSCRIPTEN_BINDINGS(my_module) {
    function("hello", &hello);
    function("add", &add);
}
```
