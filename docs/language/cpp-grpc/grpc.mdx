```shell
protoc \
  --proto_path=<protoRoot> \
  --cpp_out=<cppRoot> \
  --grpc_out=<cppRoot> \
  --plugin=protoc-gen-grpc="$(which grpc_cpp_plugin)" \
  <protoFile>
```
