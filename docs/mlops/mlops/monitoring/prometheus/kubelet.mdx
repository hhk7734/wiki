
## 설치

### kube-prometheus-stack

```yaml title="kube-prometheus-stack-values.yaml"
kubernetesServiceMonitors:
  enabled: true

kubelet:
  enabled: true

  serviceMonitor:
    additionalLabels: {}

prometheusOperator:
  enabled: true

  kubeletService:
    enabled: true
    namespace: kube-system
    # '{{ template "kube-prometheus-stack.fullname" . }}-kubelet' by default
    name: ""
```
