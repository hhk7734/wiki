## 설치

- https://hub.docker.com/_/postgres

```yaml
---
apiVersion: v1
kind: ServiceAccount
metadata:
  name: postgresql
  namespace: storage
  labels:
    app.kubernetes.io/name: postgresql
    app: postgresql
```

```yaml
---
apiVersion: v1
kind: Secret
metadata:
  name: postgresql
  namespace: storage
  labels:
    app.kubernetes.io/name: postgresql
    app: postgresql
type: Opaque
data:
  password: <base64(password)>
```

```yaml
---
apiVersion: apps/v1
kind: StatefulSet
metadata:
  name: postgresql
  namespace: storage
  labels:
    app.kubernetes.io/name: postgresql
    app: postgresql
spec:
  replicas: 1
  selector:
    matchLabels:
      app.kubernetes.io/name: postgresql
      app: postgresql
  template:
    metadata:
      name: postgresql
      labels:
        app.kubernetes.io/name: postgresql
        app: postgresql
    spec:
      serviceAccountName: postgresql
      containers:
        - name: postgresql
          image: postgres:15.4-alpine3.18
          imagePullPolicy: 'IfNotPresent'
          ports:
            - name: postgresql
              containerPort: 5432
              protocol: TCP
          env:
            - name: POSTGRES_USER
              value: postgres
            - name: POSTGRES_PASSWORD
              valueFrom:
                secretKeyRef:
                  name: postgresql
                  key: password
            - name: PGDATA
              value: /var/lib/postgresql/data/pgdata
          livenessProbe:
            failureThreshold: 6
            initialDelaySeconds: 30
            periodSeconds: 10
            successThreshold: 1
            timeoutSeconds: 5
            exec:
              command:
                - /bin/sh
                - -c
                - exec pg_isready -U "postgres" -h 127.0.0.1 -p 5432
          readinessProbe:
            failureThreshold: 6
            initialDelaySeconds: 5
            periodSeconds: 10
            successThreshold: 1
            timeoutSeconds: 5
            exec:
              command:
                - /bin/sh
                - -c
                - -e
                - |
                  exec pg_isready -U "postgres" -h 127.0.0.1 -p 5432
                  [ -f /var/lib/postgresql/data/.initialized ]
          resources:
            requests:
              cpu: 100m
              memory: 256Mi
            limits:
              memory: 256Mi
          volumeMounts:
            - name: dshm
              mountPath: /dev/shm
            - name: data
              mountPath: /var/lib/postgresql/data
      affinity:
        podAntiAffinity:
          preferredDuringSchedulingIgnoredDuringExecution:
            - podAffinityTerm:
                topologyKey: kubernetes.io/hostname
                labelSelector:
                  matchLabels:
                    app.kubernetes.io/name: postgresql
                    app: postgresql
              weight: 1
      volumes:
        - name: dshm
          emptyDir:
            medium: Memory
  volumeClaimTemplates:
    - metadata:
        name: data
        labels:
          app.kubernetes.io/name: postgresql
          app: postgresql
      spec:
        accessModes:
          - 'ReadWriteOnce'
        resources:
          requests:
            storage: '8Gi'
```

- [PersistentVolume](/docs/mlops/mlops/kubernetes/persistent-volume)

:::warning
볼륨을 `/var/lib/postgresql`에 마운트 시켰을 때, `/var/lib/postgresql/data` 경로는 로컬 볼륨을 사용하므로 이 값들을 바꾸지 않으면 데이터가 삭제됩니다.
:::

```yaml
---
apiVersion: v1
kind: Service
metadata:
  name: postgresql
  namespace: storage
  labels:
    app.kubernetes.io/name: postgresql
    app: postgresql
spec:
  type: ClusterIP
  ports:
    - name: postgresql
      port: 5432
      protocol: TCP
      targetPort: postgresql
  selector:
    app.kubernetes.io/name: postgresql
    app: postgresql
```
