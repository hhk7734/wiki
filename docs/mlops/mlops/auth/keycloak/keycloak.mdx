## 설치

```shell
helm repo add bitnami https://charts.bitnami.com/bitnami
```

```shell
helm repo update bitnami \
&& helm search repo bitnami/keycloak -l | head -n 10
```

```shell
helm show values bitnami/keycloak \
    --version 13.0.0 \
    > keycloak-values.yaml
```

```yaml title="keycloak-values.yaml"
commonLabels: {}

auth:
  adminUser: <user>
  adminPassword: <password>

resources:
  requests:
    cpu: 250m
    memory: 1Gi
  limits:
    cpu: 250m
    memory: 1Gi

podLabels: {}

service:
  type: ClusterIP

postgresql:
  enabled: false

externalDatabase:
  host: <host>
  port: <port>
  user: <user>
  password: <password>
  database: <database>

cache:
  enabled: false
```

```shell
helm template keycloak bitnami/keycloak \
    --version 13.0.0 \
    -n auth \
    -f keycloak-values.yaml \
    > keycloak.yaml
```

```shell
helm upgrade keycloak bitnami/keycloak \
    --install \
    --version 13.0.0 \
    -n auth \
    --history-max 3 \
    -f keycloak-values.yaml
```
