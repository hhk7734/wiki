## 설치

```shell
helm repo add bitnami https://charts.bitnami.com/bitnami
```

```shell
helm repo update bitnami \
&& helm search repo bitnami/keycloak -l | head -n 10
```

```shell
helm pull bitnami/keycloak --version 19.3.4
```

```shell
helm show values bitnami/keycloak --version 19.3.4 > keycloak-19.3.4.yaml
```

```yaml title="keycloak-values.yaml"
commonLabels: {}

auth:
  adminUser: <user>
  adminPassword: <password>

resources:
  requests:
    cpu: 50m
    memory: 1Gi
  limits:
    memory: 1Gi

podLabels: {}

service:
  type: ClusterIP

postgresql:
  # https://github.com/bitnami/charts/blob/main/bitnami/postgresql/values.yaml
  enabled: false
  primary:
    affinity: {}
    tolerations: []
    persistence:
      storageClass: ""

externalDatabase:
  host: <host>
  port: <port>
  user: <user>
  password: <password>
  database: <database>

cache:
  enabled: false
```

```shell
helm template keycloak keycloak-19.3.4.tgz \
    -n auth \
    -f keycloak-values.yaml \
    > keycloak.yaml
```

```shell
helm upgrade keycloak keycloak-19.3.4.tgz \
    --install \
    --history-max 5 \
    -n auth \
    -f keycloak-values.yaml
```
