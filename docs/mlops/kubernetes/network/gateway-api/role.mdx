### Route binding

```yaml
apiVersion: gateway.networking.k8s.io/v1
kind: Gateway
spec:
  listeners:
    - allowedRoutes:
        namespaces:
          from: Selector
          selector:
            - key: kubernetes.io/metadata.name
              operator: In
              values:
                - <namespace1>
                - <namespace2>
```
