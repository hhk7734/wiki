## 설치

```shell
go install github.com/pulumi/crd2pulumi@latest
```

## CRDs to yaml

```shell
for crd in $(kubectl get crd -o name); do
    case $crd in
        *istio*)
            echo $crd | cut -d "/" -f 2
            echo "---" >> crds.yaml
            kubectl get $crd -o yaml >> crds.yaml
            ;;
    esac
done
```

## yaml to pulumi

```shell
crd2pulumi \
    --nodejsPath ./crds/istio \
    --nodejsName istio \
    crds.yaml
```
